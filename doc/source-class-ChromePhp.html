262:      *
263:      * @param string $type
264:      * @return void

266:     protected static function _log(array $args)
267:     {
268:         $type = $args['type'];
269:         unset($args['type']);
270: 
271:         // nothing passed in, don't do anything
272:         if (count($args) == 0 && $type != self::GROUP_END) {
273:             return;
274:         }
275: 
276:         // default to single
277:         $label = null;
278:         $value = isset($args[0]) ? $args[0] : '';
279: 
280:         $logger = self::getInstance();
281: 
282:         if ($logger->_error_triggered) {
283:             return;
284:         }
285: 
286:         // if there are two values passed in then the first one is the label
287:         if (count($args) == 2) {
288:             $label = $args[0];
289:             $value = $args[1];
290:         }
291: 
292:         $logger->_processed = array();
293:         $value = $logger->_convert($value);
294: 
295:         $backtrace = debug_backtrace(false);
296:         $level = $logger->getSetting(self::BACKTRACE_LEVEL);
297: 
298:         $backtrace_message = 'unknown';
299:         if (isset($backtrace[$level]['file']) && isset($backtrace[$level]['line'])) {
300:             $backtrace_message = $backtrace[$level]['file'] . ' : ' . $backtrace[$level]['line'];

306:     /**
307:      * converts an object to a better format for logging
308:      *
309:      * @param Object
310:      * @return array
311:      */
312:     protected function _convert($object)
313:     {
314:         // if this isn't an object then just return it
315:         if (!is_object($object)) {
316:             return $object;
317:         }
318: 
319:         //Mark this object as processed so we don't convert it twice and it
320:         //Also avoid recursion when objects refer to each other
321:         $this->_processed[] = $object;
322: 
323:         $object_as_array = array();
324: 
325:         // first add the class name
326:         $object_as_array['class'] = get_class($object);
327: 
328:         // loop through object vars
329:         $object_vars = get_object_vars($object);
330:         foreach ($object_vars as $key => $value) {
331: 
332:             // same instance as parent object
333:             if ($value === $object || in_array($value, $this->_processed, true)) {
334:                 $value = 'recursion - parent object [' . get_class($value) . ']';
335:             }
336:             $object_as_array[$key] = $this->_convert($value);
337:         }
338: 
339:         $reflection = new ReflectionClass($object);
340: 
341:         // loop through the properties and add those
342:         foreach ($reflection->getProperties() as $property) {
343: 
344:             // if one of these properties was already added above then ignore it
345:             if (array_key_exists($property->getName(), $object_vars)) {
346:                 continue;
347:             }
348:             $type = $this->_getPropertyKey($property);
349: 
350:             if ($this->_php_version >= 5.3) {
351:                 $property->setAccessible(true);
352:             }
353: 
354:             try {
355:                 $value = $property->getValue($object);
356:             } catch (ReflectionException $e) {
357:                 $value = 'only PHP 5.3 can access private/protected properties';
358:             }
359: 
360:             // same instance as parent object
361:             if ($value === $object || in_array($value, $this->_processed, true)) {
362:                 $value = 'recursion - parent object [' . get_class($value) . ']';
363:             }
364: 
365:             $object_as_array[$type] = $this->_convert($value);
366:         }
367:         return $object_as_array;
368:     }

398:     protected function _addRow($label, $log, $backtrace, $type)
399:     {
400:         // if this is logged on the same line for example in a loop, set it to null to save space
401:         if (in_array($backtrace, $this->_backtraces)) {

413:         if ($this->getSetting(self::LOG_PATH) === null && !$this->_willFit($row)) {

421:     /**
422:      * clears existing rows in special cases
423:      *
424:      * for ajax requests chrome will be listening for cookie changes
425:      * this means we can send the cookie data one row at a time as it comes in
426:      *
427:      * @return void
428:      */
429:     protected function _clearRows()
430:     {
431:         // if we are in file mode we want the file to have all the log data
432:         if ($this->getSetting(self::LOG_PATH) !== null) {
433:             return;
434:         }
435: 
436:         // X-Requested-With header not present or not equal to XMLHttpRequest
437:         if (!isset($_SERVER['HTTP_X_REQUESTED_WITH'])) {
438:             return;
439:         }

441:         if ($_SERVER['HTTP_X_REQUESTED_WITH'] != 'XMLHttpRequest') {
442:             return;

448:     /**
449:      * determines if this row will fit in the cookie
450:      *
451:      * @param array $row
452:      * @return bool
453:      */
454:     protected function _willFit($row)
455:     {
456:         $json = $this->_json;
457:         $json['rows'][] = $row;
458: 
459:         // if we don't have multibyte string length available just use regular string length
460:         // this doesn't have to be perfect, just want to prevent sending more data
461:         // than chrome or apache can handle in a cookie
462:         $encoded_string = $this->_encode($json);
463:         $size = function_exists('mb_strlen') ? mb_strlen($encoded_string) : strlen($encoded_string);
464: 
465:         // if the size is greater than the max transfer size
466:         if ($size > $this->getSetting(self::MAX_TRANSFER)) {
467:             return false;
468:         }
469: 
470:         return true;
471:     }
